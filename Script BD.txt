create table produto(
id int primary key generated always as identity,
nomeproduto varchar(280),
nomeextenso varchar(2000),
estrelas int,
status boolean,
quantidade int,
preco double
);

create table imagens(
id int primary key generated always as identity,
dir varchar(2000),
idproduto int
);

create table usuario(
id int primary key generated always as identity,
nome varchar(280),
senha varchar(1000),
status boolean,
tipocadastro int
);

create table tipocadastro(
id int primary key generated always as identity,
descricao varchar(200)
);
